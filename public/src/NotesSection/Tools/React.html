4.6. Custom Hook

Custom hook is a user defined hook which is created from various other hooks This is a hook to get the breed list for a particular animal We have a local cache to avoid calling the api again and again It also returns a status giving us the status of the request 

import { useState, useEffect } from "react";

const localCache = {};

export default function useBreedList(animal) {
  const [breedList, setBreedList] = useState([]);
  const [status, setStatus] = useState("unloaded");

  useEffect(() => {
    if (!animal) {
      setBreedList([]);
    } else if (localCache[animal]) {
      setBreedList(localCache[animal]);
    } else {
      requestBreedList();
    }
    async function requestBreedList() {
      setBreedList([]);
      setStatus("loading");

      const res = await fetch(
        `https://pets-v2.dev-apis.com/breeds?animal=${animal}`
      );
      const json = await res.json();
      localCache[animal] = json.breeds || [];
      setBreedList(localCache[animal]);
      setStatus("loaded");
    }
  }, [animal]);
  return [breedList, status];
}

requestBreedlist is inside useEffect as it is recommended by react and it is part of one effect It must be rendered by one render Async function is a function that returns a promise To do async await inside a function we must do a asynch inside the effect Status helps us to test the program as needed e is a fake dom event or a synthetic react event It is a part of the virtual dom React manages the vritual dom and minimizes the changing then work on the actual but slow dom

import {useEffect,useState} from 'react'

function useCurrencyInfo(currency){
    const[data,setData] = useState({})
    let URL = ""
   useEffect(()=>{
       fetch(URL)
       .then((res)=>res.json())
       .then((res)=> setData(res[currency]))
   },[currency]) // Calling the API only when component is update
    return data
}

export default useCurrencyInfo;
